## 1. 入力された文字列を逆さに表示するプログラムを作成しなさい．  

    例：入力>> tamaki
        出力>> ikamat

~~~c
#include <stdio.h>
int main (void){
    char input[100];
    int len = 0;
    int i;

    printf("入力：");
    scanf("%s", input);

    while (input[len] != '\0'){
        len++;
    }

    for(i = len -1; i >= 0; i--){
        printf("%c", input);
    }
    return 0;
}
~~~

## 2. 文字列s1の後ろに文字列s2を接続するプログラムを作成しなさい．ただし，文字列操作関数は使用してはいけない．
~~~c
#include <stdio.h>
int main (void){
    char s1[100];
    char s2[10];

    // 必要な変数を宣言してください
    int len = 0;
    int i;

    // 入力受付
    printf("s1:");
    scanf("%s", s1);

    printf("s2:");
    scanf("%s", s2);

    // 接続する文を考えてください．

    while (s1[len] != '\0'){
        len++;
    }// null文字探し

    while (s2[i ]!= '\0'){
        s1[len] = s2[i];

        len++;
        i++;
    }

    s1[len] = '\0';

    printf("%s\n", s1);

    return 0;
}
~~~

## 3. 整数の高さaと，底辺の長さbを使って，三角形の面積を実数で計算する関数を作成しなさい．
